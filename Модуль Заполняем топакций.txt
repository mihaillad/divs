// <ОтправитьДляОбработки>
// Параметры:
//  <Адрес>
//  <ИмяВыходногоФайла>
//  <Метод>  - по умолчанию GET
//
Функция ОтправитьДляОбработки(ПараметрыВыполнения) Экспорт  
	
	СтруктураURI = ОбщегоНазначенияКлиентСервер.СтруктураURI(ПараметрыВыполнения.Адрес);
	Хост = СтруктураURI.Хост;
	Порт = СтруктураURI.Порт;
	Путь = СтруктураURI.ПутьНаСервере;
	Логин = СтруктураURI.Логин;
	Пароль = СтруктураURI.Пароль;
	Таймаут = 3;
	
	ИмяВыходногоФайла = "";
	Если ПараметрыВыполнения.Свойство("ИмяВыходногоФайла") Тогда ИмяВыходногоФайла = ПараметрыВыполнения.ИмяВыходногоФайла; КонецЕсли;
	
	Метод = "GET";
	Если ПараметрыВыполнения.Свойство("Метод") Тогда Метод = ПараметрыВыполнения.Метод; КонецЕсли;
	
	ТекстЗапроса = "";
	Если ПараметрыВыполнения.Свойство("ТекстЗапроса") Тогда ТекстЗапроса = ПараметрыВыполнения.ТекстЗапроса; КонецЕсли;
	
	
	СтрокаОтвет = "";
	
	СтруктураОтвета = Новый Структура("Результат,Текст,КодСостояния",Неопределено,"");
	
	Попытка
		
		Соед = Новый HTTPСоединение(Хост,Порт,Логин,Пароль,,Таймаут,,);  
		Запрос = Новый HTTPЗапрос(Путь);
		Запрос.Заголовки.Вставить("Content-Type", "application/json");
		Запрос.УстановитьТелоИзСтроки(ТекстЗапроса);  
		
		Если ВРег(Метод) = "GET" Тогда
			Ответ = Соед.Получить(Запрос, ИмяВыходногоФайла);
		ИначеЕсли ВРег(Метод) = "POST" Тогда
			Ответ = Соед.ОтправитьДляОбработки(Запрос, ИмяВыходногоФайла);
		Иначе
			Ответ = Соед.ВызватьHTTPМетод(ВРег(Метод), Запрос, ИмяВыходногоФайла);
		КонецЕсли;
		
		СтрокаОтвет = Ответ.ПолучитьТелоКакСтроку();
		СтруктураОтвета.КодСостояния = Ответ.кодСостояния;
		Если Ответ.КодСостояния <> 200 Тогда
			СтруктураОтвета.Результат = Ложь;
			СтруктураОтвета.Текст = "Код состояния "+Ответ.КодСостояния+". "+Ответ.ПолучитьТелоКакСтроку();
		Иначе
			СтруктураОтвета = лмОбщегоНазначения.ИзДжсон(СтрокаОтвет);
			Если ТипЗнч(СтруктураОтвета) = Тип("Структура") Тогда
				СтруктураОтвета.вставить("КодСостояния",200);
			КонецЕсли;
		КонецЕсли;
		
		Возврат СтруктураОтвета;
		
	Исключение

		
		СтруктураОтвета.Результат = Ложь;
		СтруктураОтвета.Текст = ОписаниеОшибки()+": "+СтрокаОтвет;
		
		Возврат СтруктураОтвета;
		
	КонецПопытки

Функция ПолучитьСпецификациюМосбиржиНаСервере(SECID)
	//https://iss.moex.com/iss/securities/RU000A106EP1.json?iss.meta=off
	
	Адрес =  "https://iss.moex.com/iss/securities/"+SECID+".json?iss.meta=off";
	ИмяФайла = ""+SECID+".json";
	СтруктураДанных = ПроверитьАктуальностьФайлаИЗагрузить(ИмяФайла,Адрес);
	Если СтруктураДанных = Неопределено Тогда
		Возврат Неопределено;
	КонецЕсли;
	
	Структура = СтруктураДанных;///на что заменить?
	
	Массив = СтруктураДанных.description.data;
	Для Каждого Элемент ИЗ Массив Цикл
		Имя = Элемент[0];
		Если Структура.Свойство(Имя) Тогда
			Структура[Имя] = Элемент[2];	
		КонецЕсли;
		
	КонецЦикла;
	
	Индекс_is_primary = СтруктураДанных.boards.columns.Найти("is_primary");
	Индекс_boardid = СтруктураДанных.boards.columns.Найти("boardid");
	Для Каждого Элемент ИЗ СтруктураДанных.boards.data Цикл
		is_primary = Элемент[Индекс_is_primary];
		Если ЗначениеЗаполнено(is_primary) = Истина 
			И is_primary = 1 Тогда
			Структура.BOARDID = Элемент[Индекс_boardid];
			Прервать;
		КонецЕсли;
		
	КонецЦикла;
	
	Возврат Структура;
